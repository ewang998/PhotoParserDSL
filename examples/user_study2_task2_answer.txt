CANVAS WIDTH 1000 HEIGHT 1000 COLOR #add8e6;
LET dog.png BE dog;
LET cat.jpg BE cat;
DECLARE ROTATEANDGREYSCALE AS ROTATE 240 AND GREYSCALE;
APPLY ROTATEANDGREYSCALE TO cat;
APPLY ROTATEANDGREYSCALE TO dog;
DRAW TO CANVAS dog AT X 0 Y 250;
DRAW TO CANVAS cat AT X 400 Y 250;
CLONE dog AS dogCopy;
APPLY INVERT TO dogCopy;
APPLY FLIP HORIZONTAL TO dogCopy;
DRAW TO CANVAS dogCopy AT X 250 Y 500;
RENDER AS dogandcat.png;
